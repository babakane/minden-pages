The Mobile-First Imperative

  • Intermittent Connectivity — Users need offline functionality
  • Limited Bandwidth — Heavy pages exclude potential customers

1. Offline-First Data Layer

JavaScript • OfflineSyncManager.js
class OfflineSyncManager {
    constructor() {
        this.db = null;
        this.syncQueue = [];
        this.isOnline = navigator.onLine;
        
        // Listen for connectivity changes
        window.addEventListener('online', () => this.handleOnline());
        window.addEventListener('offline', () => this.handleOffline());
    }
    
    async initDB() {
        // Initialize IndexedDB for local storage
        return new Promise((resolve, reject) => {
            const request = indexedDB.open('ECommerceDB', 1);
            
            request.onerror = () => reject(request.error);
            request.onsuccess = () => {
                this.db = request.result;
                resolve(this.db);
            };
            
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                
                // Object stores for offline functionality
                if (!db.objectStoreNames.contains('products')) {
                    db.createObjectStore('products', { keyPath: 'id' });
                }
                if (!db.objectStoreNames.contains('cart')) {
                    db.createObjectStore('cart', { keyPath: 'itemId' });
                }
                if (!db.objectStoreNames.contains('syncQueue')) {
                    db.createObjectStore('syncQueue', { 
                        keyPath: 'id', 
                        autoIncrement: true 
                    });
                }
            };
        });
    }
    
    async cacheProducts(products) {
        // Store products for offline browsing
        const transaction = this.db.transaction(['products'], 'readwrite');
        const store = transaction.objectStore('products');
        
        for (const product of products) {
            await store.put(product);
        }
        
        return new Promise((resolve, reject) => {
            transaction.oncomplete = resolve;
            transaction.onerror = () => reject(transaction.error);
        });
    }
    
    async getCachedProducts(category = null) {
        // Retrieve products when offline
        return new Promise((resolve, reject) => {
            const transaction = this.db.transaction(['products'], 'readonly');
            const store = transaction.objectStore('products');
            const request = store.getAll();
            
            request.onsuccess = () => {
                let products = request.result;
                if (category) {
                    products = products.filter(p => p.category === category);
                }
                resolve(products);
            };
            request.onerror = () => reject(request.error);
        });
    }
    
    async queueAction(action) {
        // Queue actions for sync when online
        const transaction = this.db.transaction(['syncQueue'], 'readwrite');
        const store = transaction.objectStore('syncQueue');
        
        await store.add({
            action: action,
            timestamp: new Date().toISOString(),
            retries: 0
        });
        
        if (this.isOnline) {
            this.processSyncQueue();
        }
    }
    
    async processSyncQueue() {
        // Process pending actions when connectivity restored
        const transaction = this.db.transaction(['syncQueue'], 'readonly');
        const store = transaction.objectStore('syncQueue');
        const request = store.getAll();
        
        request.onsuccess = async () => {
            const pendingActions = request.result;
            
            for (const item of pendingActions) {
                try {
                    await this.syncAction(item.action);
                    await this.removeFromQueue(item.id);
                } catch (error) {
                    console.error('Sync failed for action:', item.id, error);
                    if (item.retries < 3) {
                        await this.incrementRetries(item.id);
                    }
                }
            }
        };
    }
    
    handleOnline() {
        this.isOnline = true;
        this.processSyncQueue();
        // Show notification to user
        this.showNotification('Back online. Syncing...');
    }
    
    handleOffline() {
        this.isOnline = false;
        this.showNotification('You are offline. Changes will sync when connected.');
    }
    
    showNotification(message) {
        // Implement toast notification
        if ('Notification' in window && Notification.permission === 'granted') {
            new Notification('E-Commerce App', { body: message });
        }
    }
}

2. Low-Bandwidth Image Optimization

JavaScript • ImageOptimizer.js
class AdaptiveImageLoader {
    constructor() {
        this.connection = navigator.connection || 
                         navigator.mozConnection || 
                         navigator.webkitConnection;
    }
    
    getImageQuality() {
        // Adapt quality based on connection
        if (!this.connection) return 'medium';
        
        const effectiveType = this.connection.effectiveType;
        
        switch(effectiveType) {
            case '4g': return 'high';
            case '3g': return 'medium';
            case '2g':
            case 'slow-2g': return 'low';
            default: return 'medium';
        }
    }
    
    getImageUrl(baseUrl, width) {
        const quality = this.getImageQuality();
        
        // Generate responsive image URL
        // Example: /api/images/product-123?w=400&q=medium&format=webp
        return `${baseUrl}?w=${width}&q=${quality}&format=webp`;
    }
    
    async lazyLoadImage(imgElement, src) {
        // Use Intersection Observer for lazy loading
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    this.loadImage(imgElement, src);
                    observer.unobserve(imgElement);
                }
            });
        });
        
        observer.observe(imgElement);
    }
    
    loadImage(imgElement, src) {
        // Add blur placeholder while loading
        imgElement.classList.add('loading');
        
        const img = new Image();
        img.onload = () => {
            imgElement.src = src;
            imgElement.classList.remove('loading');
        };
        img.src = src;
    }
}

3. Last-Mile Logistics Optimizer

Python • logistics_optimizer.py
from typing import List, Dict, Tuple
from dataclasses import dataclass
import numpy as np
from scipy.optimize import linear_sum_assignment

@dataclass
class DeliveryPoint:
    id: str
    lat: float
    lng: float
    demand: int
    time_window: Tuple[int, int]  # (start, end) in hours

class LastMileOptimizer:
    """
    Optimizes delivery routes for emerging market logistics.
    Handles crowdsourced delivery agents and complex urban terrain.
    """
    
    def __init__(self, depot_location: Tuple[float, float]):
        self.depot = depot_location
        self.agents = []
        self.delivery_points = []
    
    def calculate_distance(self, p1: Tuple[float, float], 
                          p2: Tuple[float, float]) -> float:
        """
        Calculate approximate distance between coordinates.
        Uses Haversine formula for accuracy.
        """
        R = 6371  # Earth radius in km
        
        lat1, lng1 = np.radians(p1)
        lat2, lng2 = np.radians(p2)
        
        dlat = lat2 - lat1
        dlng = lng2 - lng1
        
        a = np.sin(dlat/2)**2 + np.cos(lat1) * np.cos(lat2) * np.sin(dlng/2)**2
        c = 2 * np.arcsin(np.sqrt(a))
        
        return R * c
    
    def assign_deliveries_to_agents(self, 
                                    agents: List[Dict],
                                    deliveries: List[DeliveryPoint]) -> Dict:
        """
        Assign deliveries to available agents using Hungarian algorithm.
        Considers agent capacity, location, and delivery time windows.
        """
        if not agents or not deliveries:
            return {}
        
        # Create cost matrix: distance from agent to delivery
        cost_matrix = np.zeros((len(agents), len(deliveries)))
        
        for i, agent in enumerate(agents):
            agent_location = (agent['lat'], agent['lng'])
            for j, delivery in enumerate(deliveries):
                distance = self.calculate_distance(
                    agent_location, 
                    (delivery.lat, delivery.lng)
                )
                
                # Add penalty for time window violations
                time_penalty = self.calculate_time_penalty(
                    agent, delivery
                )
                
                cost_matrix[i][j] = distance + time_penalty
        
        # Solve assignment problem
        row_ind, col_ind = linear_sum_assignment(cost_matrix)
        
        assignments = {}
        for i, j in zip(row_ind, col_ind):
            agent_id = agents[i]['id']
            if agent_id not in assignments:
                assignments[agent_id] = []
            assignments[agent_id].append(deliveries[j])
        
        return assignments
    
    def optimize_route(self, points: List[DeliveryPoint]) -> List[DeliveryPoint]:
        """
        Optimize single agent's route using nearest neighbor heuristic
        with 2-opt improvement.
        """
        if not points:
            return []
        
        # Start from depot
        route = []
        unvisited = set(range(len(points)))
        current = self.depot
        
        while unvisited:
            # Find nearest unvisited point
            nearest = min(unvisited, 
                         key=lambda i: self.calculate_distance(
                             current, 
                             (points[i].lat, points[i].lng)
                         ))
            route.append(points[nearest])
            current = (points[nearest].lat, points[nearest].lng)
            unvisited.remove(nearest)
        
        # Apply 2-opt improvement
        return self.two_opt_improvement(route)
    
    def two_opt_improvement(self, route: List[DeliveryPoint]) -> List[DeliveryPoint]:
        """
        Improve route using 2-opt swap heuristic.
        """
        improved = True
        best = route.copy()
        
        while improved:
            improved = False
            for i in range(1, len(route) - 1):
                for j in range(i + 1, len(route)):
                    # Calculate current segment cost
                    current_cost = (
                        self.calculate_distance(
                            (best[i-1].lat, best[i-1].lng),
                            (best[i].lat, best[i].lng)
                        ) +
                        self.calculate_distance(
                            (best[j-1].lat, best[j-1].lng),
                            (best[j].lat, best[j].lng) if j < len(best) else self.depot
                        )
                    )
                    
                    # Calculate cost if reversed
                    new_cost = (
                        self.calculate_distance(
                            (best[i-1].lat, best[i-1].lng),
                            (best[j-1].lat, best[j-1].lng)
                        ) +
                        self.calculate_distance(
                            (best[i].lat, best[i].lng),
                            (best[j].lat, best[j].lng) if j < len(best) else self.depot
                        )
                    )
                    
                    if new_cost < current_cost:
                        # Apply 2-opt swap
                        best[i:j] = reversed(best[i:j])
                        improved = True
        
        return best
    
    def calculate_time_penalty(self, agent: Dict, delivery: DeliveryPoint) -> float:
        """
        Calculate penalty for time window violations.
        """
        # Simplified: assume agent starts at 9 AM
        estimated_arrival = 9 + (self.calculate_distance(
            (agent['lat'], agent['lng']),
            (delivery.lat, delivery.lng)
        ) / 20)  # Assume 20 km/h average
        
        start, end = delivery.time_window
        
        if estimated_arrival < start:
            return (start - estimated_arrival) * 0.5  # Waiting penalty
        elif estimated_arrival > end:
            return (estimated_arrival - end) * 2.0  # Late penalty
        
        return 0


# Example usage
if __name__ == "__main__":
    optimizer = LastMileOptimizer(depot_location=(14.7167, -17.4677))
    
    # Sample delivery points
    deliveries = [
        DeliveryPoint("D1", 14.75, -17.45, 2, (10, 14)),
        DeliveryPoint("D2", 14.70, -17.50, 1, (9, 12)),
        DeliveryPoint("D3", 14.72, -17.48, 3, (14, 18)),
    ]
    
    # Sample agents
    agents = [
        {"id": "A1", "lat": 14.71, "lng": -17.46, "capacity": 5},
        {"id": "A2", "lat": 14.73, "lng": -17.44, "capacity": 4},
    ]
    
    # Optimize assignments
    assignments = optimizer.assign_deliveries_to_agents(agents, deliveries)
    
    for agent_id, points in assignments.items():
        optimized_route = optimizer.optimize_route(points)
        print(f"Agent {agent_id}: {len(optimized_route)} deliveries")
        for i, point in enumerate(optimized_route):
            print(f"  {i+1}. {point.id} at ({point.lat}, {point.lng})")

💡 Key Insight: Progressive Web Apps

For emerging markets, consider building a Progressive Web App (PWA) instead of native mobile apps. PWAs offer:

• Offline functionality via Service Workers
• Smaller download size than native apps
• No app store approval process
• Works on low-end devices
• Automatic updates

Business Model Integration

The technical architecture supports multiple business models common in emerging markets:

  • Cash on Delivery (COD): Mobile app supports COD workflows with delivery agent verification
  • Agent Network: Logistics optimizer supports crowdsourced delivery agents
  • Micro-fulfillment: Small inventory caches distributed across urban areas
  • Group Buying: Offline sync enables community-based bulk ordering
