2. Conservation Status Tracker

from dataclasses import dataclass
from datetime import datetime
from typing import List, Dict
import json

@dataclass
class SpeciesRecord:
    species_name: str
    iucn_status: str  # CR, EN, VU, NT, LC
    population_estimate: int
    last_sighting: datetime
    location: Dict[str, float]  # lat, lng
    habitat_quality: float  # 0-100
    threats: List[str]
    
    def conservation_priority(self) -> int:
        """Calculate priority score 1-10 (10 = highest priority)."""
        score = 0
        
        # IUCN status weight
        iucn_weights = {'CR': 5, 'EN': 4, 'VU': 3, 'NT': 2, 'LC': 1}
        score += iucn_weights.get(self.iucn_status, 0)
        
        # Population weight (smaller populations = higher priority)
        if self.population_estimate < 100:
            score += 3
        elif self.population_estimate < 1000:
            score += 2
        elif self.population_estimate < 10000:
            score += 1
            
        # Habitat quality (lower quality = higher priority)
        if self.habitat_quality < 30:
            score += 2
        elif self.habitat_quality < 60:
            score += 1
            
        return min(score, 10)

class ConservationDashboard:
    """Track and visualize conservation metrics."""
    
    def __init__(self):
        self.species_records = {}
        self.sightings = []
        
    def register_species(self, record: SpeciesRecord):
        """Add species to monitoring system."""
        self.species_records[record.species_name] = record
        
    def record_sighting(self, species: str, location: Dict,
                       count: int = 1, image_ref: str = None):
        """Log wildlife sighting."""
        sighting = {
            'species': species,
            'timestamp': datetime.now().isoformat(),
            'location': location,
            'count': count,
            'image': image_ref
        }
        self.sightings.append(sighting)
        
        # Update last sighting
        if species in self.species_records:
            self.species_records[species].last_sighting = datetime.now()
            
    def get_conservation_summary(self) -> Dict:
        """Generate conservation status dashboard."""
        summary = {
            'total_species': len(self.species_records),
            'critically_endangered': 0,
            'endangered': 0,
            'vulnerable': 0,
            'total_sightings_30d': 0,
            'priority_actions': []
        }
        
        for name, record in self.species_records.items():
            if record.iucn_status == 'CR':
                summary['critically_endangered'] += 1
            elif record.iucn_status == 'EN':
                summary['endangered'] += 1
            elif record.iucn_status == 'VU':
                summary['vulnerable'] += 1
                
            # Check for priority actions needed
            if record.conservation_priority() >= 7:
                summary['priority_actions'].append({
                    'species': name,
                    'priority': record.conservation_priority(),
                    'action': 'Immediate intervention required'
                })
        
        # Recent sightings
        from datetime import timedelta
        cutoff = datetime.now() - timedelta(days=30)
        summary['total_sightings_30d'] = len([
            s for s in self.sightings 
            if datetime.fromisoformat(s['timestamp']) > cutoff
        ])
        
        return summary
