1. IoT Data Collection Layer

Python • iot_data_collector.py
import asyncio
import json
from datetime import datetime
from typing import Dict, List
import aiohttp

class IoTResourceCollector:
    """Collects sensor data from distributed IoT networks."""
    
    def __init__(self, endpoints: List[str]):
        self.endpoints = endpoints
        self.buffer = []
        
    async def collect_sensor_data(self, sensor_id: str) -> Dict:
        """Fetch data from a single sensor endpoint."""
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get(f"{self.endpoints}/{sensor_id}") as resp:
                    if resp.status == 200:
                        data = await resp.json()
                        return {
                            'sensor_id': sensor_id,
                            'timestamp': datetime.now().isoformat(),
                            'value': data.get('reading'),
                            'unit': data.get('unit'),
                            'location': data.get('coordinates')
                        }
        except Exception as e:
            return {'sensor_id': sensor_id, 'error': str(e)}
    
    async def batch_collect(self, sensor_ids: List[str]) -> List[Dict]:
        """Collect data from multiple sensors concurrently."""
        tasks = [self.collect_sensor_data(sid) for sid in sensor_ids]
        return await asyncio.gather(*tasks)

2. Forecasting Engine

Python • forecast_engine.py
import pandas as pd
import numpy as np
from statsmodels.tsa.arima.model import ARIMA
from prophet import Prophet
from sklearn.metrics import mean_absolute_error

class ResourceForecaster:
    """Time-series forecasting for resource demand prediction."""
    
    def __init__(self, model_type: str = 'prophet'):
        self.model_type = model_type
        self.model = None
        
    def prepare_data(self, df: pd.DataFrame) -> pd.DataFrame:
        """Prepare time-series data for modeling."""
        df['ds'] = pd.to_datetime(df['timestamp'])
        df['y'] = df['value'].astype(float)
        return df[['ds', 'y']].sort_values('ds')
    
    def fit(self, df: pd.DataFrame) -> 'ResourceForecaster':
        """Train forecasting model."""
        data = self.prepare_data(df)
        
        if self.model_type == 'prophet':
            self.model = Prophet(
                yearly_seasonality=True,
                weekly_seasonality=True,
                daily_seasonality=True
            )
            self.model.fit(data)
        elif self.model_type == 'arima':
            self.model = ARIMA(data['y'], order=(5, 1, 0))
            self.model = self.model.fit()
            
        return self
    
    def predict(self, periods: int = 30) -> pd.DataFrame:
        """Generate future predictions."""
        if self.model_type == 'prophet':
            future = self.model.make_future_dataframe(periods=periods)
            forecast = self.model.predict(future)
            return forecast[['ds', 'yhat', 'yhat_lower', 'yhat_upper']]
        else:
            forecast = self.model.forecast(steps=periods)
            return pd.DataFrame({'yhat': forecast})
