import random
from dataclasses import dataclass
from datetime import datetime
from typing import List, Dict
import json

@dataclass
class SoilReading:
    sensor_id: str
    timestamp: datetime
    moisture: float  # percentage
    ph: float
    nitrogen: float  # mg/kg
    phosphorus: float
    potassium: float
    temperature: float  # celsius
    
    def to_dict(self) -> Dict:
        return {
            'sensor_id': self.sensor_id,
            'timestamp': self.timestamp.isoformat(),
            'moisture': self.moisture,
            'ph': self.ph,
            'nitrogen': self.nitrogen,
            'phosphorus': self.phosphorus,
            'potassium': self.potassium,
            'temperature': self.temperature,
            'health_score': self.calculate_health()
        }
    
    def calculate_health(self) -> float:
        """Calculate soil health score 0-100."""
        score = 100
        
        # Optimal ranges
        if not (6.0 <= self.ph <= 7.5):
            score -= 15
        if self.moisture < 30:
            score -= 20
        if self.nitrogen < 20:
            score -= 10
        if self.phosphorus < 15:
            score -= 10
            
        return max(0, score)

class SoilMonitoringNetwork:
    """Manages distributed soil sensor network."""
    
    def __init__(self):
        self.sensors = {}
        self.readings_history = []
    
    def register_sensor(self, sensor_id: str, location: Dict):
        """Register new soil sensor in the network."""
        self.sensors[sensor_id] = {
            'location': location,
            'installed': datetime.now(),
            'status': 'active'
        }
    
    def record_reading(self, reading: SoilReading):
        """Store sensor reading and trigger alerts if needed."""
        self.readings_history.append(reading.to_dict())
        
        # Alert on low health
        if reading.calculate_health() < 50:
            self.send_alert(reading.sensor_id, 'low_soil_health')
    
    def get_field_health_report(self) -> Dict:
        """Generate aggregated soil health report."""
        if not self.readings_history:
            return {'status': 'no_data'}
        
        recent = self.readings_history[-100:]  # Last 100 readings
        health_scores = [r['health_score'] for r in recent]
        
        return {
            'average_health': sum(health_scores) / len(health_scores),
            'sensors_active': len(self.sensors),
            'total_readings': len(self.readings_history),
            'alerts_24h': len([r for r in recent if r['health_score'] < 50]),
            'recommendations': self.generate_recommendations(recent)
        }
    
    def generate_recommendations(self, readings: List[Dict]) -> List[str]:
        """Generate farming recommendations based on data."""
        recs = []
        
        avg_moisture = sum(r['moisture'] for r in readings) / len(readings)
        avg_ph = sum(r['ph'] for r in readings) / len(readings)
        
        if avg_moisture < 30:
            recs.append("Increase irrigation - low soil moisture detected")
        elif avg_moisture > 70:
            recs.append("Reduce irrigation - risk of waterlogging")
            
        if avg_ph < 6.0:
            recs.append("Apply lime - soil pH too acidic")
        elif avg_ph > 7.5:
            recs.append("Apply sulfur - soil pH too alkaline")
            
        return recs

from datetime import datetime, timedelta
from typing import Optional

class SmartIrrigationController:
    """Automated irrigation based on soil conditions and weather."""
    
    def __init__(self, field_id: str, valve_pins: Dict[str, int]):
        self.field_id = field_id
        self.valve_pins = valve_pins
        self.irrigation_log = []
        self.target_moisture = 45  # percent
        
    def should_irrigate(self, current_moisture: float,
                        weather_forecast: Dict) -> bool:
        """Determine if irrigation is needed."""
        # Don't irrigate if rain expected
        if weather_forecast.get('rain_probability', 0) > 70:
            return False
            
        # Irrigate if soil too dry
        if current_moisture < self.target_moisture - 15:
            return True
            
        return False
    
    def calculate_duration(self, current_moisture: float,
                          soil_type: str = 'loam') -> int:
        """Calculate irrigation duration in minutes."""
        deficit = self.target_moisture - current_moisture
        
        # Soil absorption rates
        rates = {'sand': 2.0, 'loam': 1.0, 'clay': 0.5}
        rate = rates.get(soil_type, 1.0)
        
        return int((deficit / rate) * 2)  # minutes
    
    def execute_irrigation(self, zone: str, duration: int):
        """Trigger irrigation for specific zone."""
        if zone not in self.valve_pins:
            raise ValueError(f"Unknown zone: {zone}")
        
        # Log the action
        self.irrigation_log.append({
            'zone': zone,
            'duration': duration,
            'started': datetime.now(),
            'valve_pin': self.valve_pins[zone]
        })
        
        print(f"Irrigating zone {zone} for {duration} minutes")
        
    def get_water_usage_report(self, days: int = 7) -> Dict:
        """Calculate water consumption over period."""
        cutoff = datetime.now() - timedelta(days=days)
        recent = [i for i in self.irrigation_log if i['started'] > cutoff]
        
        total_minutes = sum(i['duration'] for i in recent)
        # Estimate liters (assume 10L per minute per zone)
        total_liters = total_minutes * 10
        
        return {
            'period_days': days,
            'total_irrigations': len(recent),
            'total_minutes': total_minutes,
            'estimated_liters': total_liters,
            'zones_used': list(set(i['zone'] for i in recent))
        }
